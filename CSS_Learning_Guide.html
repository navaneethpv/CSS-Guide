1.1 CSS Selectors

/* Element Selector */
p { color: blue; }

1.3 Colors & Backgrounds

üü° LEVEL 2: INTERMEDIATE (Layout)

2.1 Display Properties

.display-block { display: block; }        /* Full width, new line */
.display-inline { display: inline; }        /* Side by side, no width */
.display-inline-block { display: inline-block; } /* Both properties */
.display-flex { display: flex; }            /* Flexible layout */
.display-grid { display: grid; }            /* Grid layout */
.display-none { display: none; }            /* Hidden, no space taken */
.display-hidden { visibility: hidden; }     /* Hidden, space taken */

2.2 Flexbox (Most Used!)

.flex-container {
    display: flex;
    justify-content: center;      /* Horizontal alignment */
    align-items: center;          /* Vertical alignment */
    gap: 1rem;                    /* Space between items */
    flex-wrap: wrap;              /* Wrap items to next line */
    flex-direction: row;          /* row, column, row-reverse, column-reverse */
}

.flex-item {
    flex: 1;                      /* Equal space */
    flex-basis: 200px;            /* Base size */
}

2.4 Positioning

.position-static { position: static; }      /* Default, follows flow */
.position-relative { position: relative; }    /* Relative to normal position */
.position-absolute { position: absolute; }    /* Relative to parent */
.position-fixed { position: fixed; }          /* Relative to viewport */
.position-sticky { position: sticky; }        /* Hybrid relative+fixed */

.positioned {
    position: absolute;
    top: 20px;                                /* Distance from top */
    left: 20px;                               /* Distance from left */
    z-index: 10;                              /* Stack order */
}

3.1 Transitions & Animations

/* Transitions (Property Change) */
.button {
    background: blue;
    transition: background 0.3s ease;        /* Which property, duration, easing */

.button:hover {
    background: red;
}

/* Animations (Keyframe) */
@keyframes slide {
    0% { transform: translateX(0); }
    50% { transform: translateX(50px); }
    100% { transform: translateX(100px); }
}

.animated {
    animation: slide 2s ease infinite;       /* Name, duration, easing, repeat */
}

3.2 Transforms

.transform {
    transform: translateX(50px);       /* Move X, Y, Z */
    transform: rotateZ(45deg);         /* Rotate */
    transform: scale(1.5);             /* Enlarge/Shrink */
    transform: skewX(10deg);           /* Skew */
    
    /* Combine multiple */
    transform: translate(50px, 30px) rotate(45deg) scale(1.2);
    
    /* 3D transforms */
    transform: rotateY(60deg);
    perspective: 1000px;               /* 3D depth */
}

3.3 Shadows & Filters

/* Text Shadow */
.text-shadow {
    text-shadow: 2px 2px 4px rgba(0,0,0,0.5); /* X, Y, blur, color */
}

/* Box Shadow */
.box-shadow {
    box-shadow: 0 10px 30px rgba(0,0,0,0.3);
    /* Inset shadow */
    box-shadow: inset 0 0 10px rgba(0,0,0,0.2);
}

/* Filters */
.filter {
    filter: blur(5px);
    filter: brightness(1.2);
    filter: contrast(1.5);
    filter: grayscale(50%);
    filter: hue-rotate(90deg);
    filter: saturate(1.5);
    filter: drop-shadow(0 0 10px rgba(0,0,0,0.5));
}

3.4 Responsive Design

/* Mobile First Approach */
.container {
    width: 100%;                       /* 100% on mobile */
    padding: 1rem;
}

/* Tablet */
@media (min-width: 768px) {
    .container {
        width: 90%;
        max-width: 700px;
    }
}

/* Desktop */
@media (min-width: 1024px) {
    .container {
        max-width: 1000px;
    }
}

/* Dark Mode */
@media (prefers-color-scheme: dark) {
    body {
        background: #1a1a1a;
        color: #fff;
    }
}

3.5 Advanced Selectors

/* Pseudo-classes */
a:hover { color: red; }                /* Mouse over */
a:active { color: blue; }              /* Being clicked */
li:first-child { margin-top: 0; }      /* First element */
li:last-child { margin-bottom: 0; }    /* Last element */
li:nth-child(2n) { background: gray; } /* Even items */
input:focus { border: 2px solid blue; } /* Focused element */

/* Pseudo-elements */
p::before { content: "‚Üí "; }           /* Before content */
p::after { content: " ‚Üê"; }            /* After content */
p::first-letter { font-size: 2rem; }   /* First letter */
p::selection { background: yellow; }   /* Selected text */

/* Attribute selectors */
input[disabled] { opacity: 0.5; }
a[href^="https"] { color: green; }     /* Starts with */
a[href$=".pdf"] { color: red; }        /* Ends with */
a[href*="example"] { color: blue; }    /* Contains */

3.6 CSS Variables (Custom Properties)

:root {
    --primary-color: #667eea;
    --secondary-color: #764ba2;
    --border-radius: 8px;
    --transition: 0.3s ease;
}

.button {
    background: var(--primary-color);
    border-radius: var(--border-radius);
    transition: all var(--transition);
}

.button:hover {
    background: var(--secondary-color);
}

/* Change variables in responsive */
@media (max-width: 768px) {
    :root {
        --border-radius: 4px;
    }
}

3.7 Performance Tips

‚ùå Slow ‚úÖ Fast Reason
width, height change transform: scale() No layout recalculation
left, top animation transform: translate() GPU accelerated
box-shadow on click Use filter: drop-shadow() Better performance
Animate 100 properties Use transform/opacity Only 2 properties trigger repaint

üü£ LEVEL 4: EXPERT TIPS

4.1 BEM Naming Convention

/* Block__Element--Modifier */
.card { }                          /* Block */
.card__header { }                  /* Element */
.card__header--active { }          /* Modifier */

/* Good for large projects */
.button { }
.button__text { }
.button--primary { }
.button--disabled { }

4.2 CSS Preprocessors (SCSS/SASS)

/* Variables */
$primary-color: #667eea;
$spacing: 1rem;

/* Nesting */
.navbar {
    background: $primary-color;
    
    &:hover {
        background: darken($primary-color, 10%);
    }
    
    .menu {
        display: flex;
        gap: $spacing;
    }
}

/* Mixins (Reusable code) */
@mixin flex-center {
    display: flex;
    justify-content: center;
    align-items: center;
}

.container {
    @include flex-center;
}
